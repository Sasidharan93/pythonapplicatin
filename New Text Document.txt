pipeline {
    agent any

    environment {
        DOCKER_HUB_USER = 'sasidharan93'
        DOCKER_HUB_PASS = 'Yuvanian@93'
        IMAGE_NAME = 'sasidharan93/pythonapplication'
        IMAGE_TAG = 'latest'
        CONTAINER_NAME = 'python-app'
        GIT_REPO = 'https://github.com/Sasidharan93/pythonapplicatin.git'  
        GIT_BRANCH = 'main'  
    }

    stages {
        stage('Clone Repository') {
            steps {
                script {
                    sh "rm -rf repo && git clone -b ${GIT_BRANCH} ${GIT_REPO} repo"
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                script {
                    sh "cd repo && docker build -t ${IMAGE_NAME}:${IMAGE_TAG} ."
                }
            }
        }

        stage('Login to Docker Hub') {
            steps {
                script {
                    sh "echo ${DOCKER_HUB_PASS} | docker login -u ${DOCKER_HUB_USER} --password-stdin"
                }
            }
        }

        stage('Push Docker Image') {
            steps {
                script {
                    sh "docker push ${IMAGE_NAME}:${IMAGE_TAG}"
                }
            }
        }

        stage('Run Python Code in Docker Container') {
            steps {
                script {
                    sh "docker run --rm --name ${CONTAINER_NAME} ${IMAGE_NAME}:${IMAGE_TAG}"
                }
            }
        }

        stage('Cleanup') {
            steps {
                script {
                    sh "docker rmi ${IMAGE_NAME}:${IMAGE_TAG}"
                    sh "docker system prune -f"
                }
            }
        }
    }

    post {
        always {
            cleanWs()  // Clean Jenkins workspace
        }
    }
}
